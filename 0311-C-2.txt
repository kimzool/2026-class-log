// Online C compiler to run C program online
#include <stdio.h>

int main() {

    int num;
    
    printf("숫자를 입력하세요..>>");
    scanf("%d",&num);
    
    //조건문
    if(num % 2 == 0) { // num이 짝수면 아래 실행
        printf("짝수\n");
        printf("2로 나눈 나머지가 0\n");
    }
    else {
        printf("홀수\n");
        printf("2로 나눈 나머지가 1\n");
    }
    


    return 0;
}
