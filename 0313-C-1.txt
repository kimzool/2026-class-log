#include <stdio.h>

int main()
{

    int menu;
    
    printf("좋아하는 메뉴를 선택해주세요 ( 1:치킨 2:피자 3:떡볶이 4:햄버거 )..>>");
    scanf("%d",&menu);
    
    
    switch(menu) {
        
        case 1:
            printf("치킨을 선택하셨습니다");
            break;
        case 2:
            printf("피자를 선택하셨습니다");
            break;
        case 3:
            printf("떡볶이를 선택하셨습니다");
            break;
        case 4:
            printf("햄버거를 선택하셨습니다.");
            break;
        default:
            printf("입력오류");
            break;
        
    }

        
    

    return 0;
}
